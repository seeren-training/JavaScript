Chapitre 09: Modifier le DOM

Modifier l'arborescence

Après avoir étudié les propriétés et méthodes pour accéder à des éléments du DOM il est envisageable de les modifier. La création et la suppression d'éléments puis la modification de valeurs et d'attributs sont étudiés dans ce chapitre.

Créer un élément

Ajouter un élément

Quand un élément est créé il ne fait pas encore partie du DOM. Après avoir créé un élément il faut l'injecter dans le document à l'aide de diverses méthodes en nous appuyant souvent sur la navigation entre les nœuds pour cibler précisément son emplacement.

createElement

La méthode createElement de document crée un nœud de type élément vide, il renvoie l'élément créé pour le tag précisé en paramètre ou lance une erreur de type InvalidCharacterError si le tag n'est pas valide.

var noeud = document.createElement(tag);
createElement var balise = document.createElement("h1");

// <h1>
 console.log(balise);
createTextNode

La méthode createTextNode de document crée un nœud de type texte, il renvoie l'élément créé pour le texte fourni en paramètre ou lance une erreur de type TypeError si aucun paramètre n'est fourni ou ne peut être converti en chaine.

var noeud = document.createTextNode(texte);
createTextNode  var texte = document.createTextNode("Hello");

// <TextNode textContent="Hello">
 console.log(texte);
appendChild

La méthode appendChild de document ou d'un élément intègre le nœud passé en paramètre à l'élément dont la méthode est invoquée et renvoie l'élément ajouté. Le nœud est ajouté à la fin de la liste des nœuds de l'élément. Si le paramètre fourni n'est pas un nœud une erreur de type TypeError sera levée.

var noeud = element.appendChild(noeud);
appendChild var tag = "h1";
var balise = document.createElement(tag);
var texte = balise.appendChild(document.createTextNode("Hello")); 

document.body.appendChild(balise);

var collection = document.getElementsByTagName(tag);

for (var i = 0, length = collection.length; i < length; i++) {
    // Hello
    console.log(collection[i].firstChild.nodeValue);
}

Dans l'exemple ci dessus, une nœud de type élément est créé puis affecté à la variable balise avec la méthode createElement puis un nœud de type texte est créé avec la méthode createTextNode puis affecté à la variable texte avec la méthode appendChild invoquée depuis la variable balise. Cette balise est injectée en dernier nœud du body avec la méthode appendChild. Une itération parcourt les tags h1 du document et logge la valeur de leur premier enfant via sa propriété nodeValue.

insertBefore

La méthode insertBefore de document ou d'un élément intègre le nœud passé en paramètre avant l'élément passé en second paramètre et renvoie l'élément ajouté. Si le paramètre fourni n'est pas un nœud une erreur de type TypeError sera levée. Si le nœud n'est pas trouvé une erreur de type NotFoundError sera levée.

var noeud = element.insertBefore(noeud, referent)
insertBefore var nodes = document.body.childNodes;

for (var i = 0, length = nodes.length; i < length; i++) {
    if (nodes[i].firstChild) {
        nodes[i].insertBefore(
            document.createTextNode("(Noeud " + i + ")"),
            nodes[i].firstChild)
    }
}

Dans l'exemple ci dessus, un nœud de type texte est créé et inséré avant le premier nœud enfant de chaque nœud enfant du body contenant un enfant.

innerHTML

La propriété innerHTML contient la représentation du contenu HTML d'un élément sous forme de chaine de caractères, il est possible d'affecter à cette propriété du contenu brut et d'injecter du contenu HTML sans passer par la gestion des nœuds.

var balisage = element.innerHTML;
innerHTML document.body.innerHTML = "\n"
                        + "<h1>Titre</h1>"
                        + "\n"
                        + "<p>Paragraphe</p>";

// <h1>
console.log(document.body.firstChild.nextSibling);

Supprimer un élément

removeChild

La méthode removeChild d'un nœud supprime un enfant passé en paramètre et renvoie cet enfant. Si le paramètre n'est pas un noeud enfant, une erreur de type NotFoundError sera levée.

var enfant = element.removeChild(enfant);
removeChild document.body.innerHTML = "<div>"
                        + "<p>Paragraphe</p>"
                        + "</div>";
                        
var removed = document.body.firstChild.removeChild(document.body.firstChild.firstChild)

// <div></div>
console.log(document.body.innerHTML);

// Paragraphe
console.log(removed.innerHTML);
replaceChild

La méthode replaceChild d'un nœud remplace un enfant passé en second paramètre par le nœud passé en premier paramètre et renvoie l'enfant remplacé. Si le premier paramètre n'est pas un nœud ou le second paramètre n'est pas un enfant, une erreur de type NotFoundError sera levée.

var enfant = element.replaceChild(noeud, enfant);
replaceChild document.body.innerHTML = "<div>"
                        + "<p>Paragraphe</p>"
                        + "</div>";
                        
var replaced = document.body.firstChild.replaceChild(
                   document.createTextNode("Texte"),
                   document.body.firstChild.firstChild)

// <div>Texte</div>
console.log(document.body.innerHTML);

// Paragraphe
console.log(replaced.innerHTML);

Modifier les attributs d'un élément

Attributs

Les attributs d'un élément peuvent êtres modifiés dynamiquement. Des méthodes permettent d'accéder aux attributs et de les modifier.

getAttribute

La méthode getAttribute d'un élément renvoie la valeur de l'attribut spécifié en paramètre ou null si l'attribut n'est pas défini.

var attribut = element.getAttribute(nom);

Soit le document suivant:

<html>
    <head>
        <script type="text/javascript" defer src="script.js"></script>
    </head>
    <body>
        <form>
            <input type="text" id="champ">
        </form>
    </body>
</html>
getAttribute var champ = document.getElementById("champ");

// champ
console.log(champ.getAttribute("id"));
setAttribute

La méthode setAttribute affecte à un élément l'attribut dont le nom est passé en premier paramètre la valeur passée en second paramètre et renvoie undefined.

element.setAttribute(nom, valeur);
setAttribute var champ = document.getElementById("champ");

champ.setAttribute("placeholder", "Saisie");

// Saisie
console.log(champ.getAttribute("placeholder"));
removeAttribute

La méthode removeAttribute supprime l'attribut dont le nom est passé en paramètre et renvoie undefined.

element.removeAttribute(nom);
removeAttribute var champ = document.getElementById("champ");

champ.removeAttribute("id");

// null
console.log(champ.getAttribute("id"));
Accesseur de propriété

Le point permet d’accéder directement à une propriété, certains mots étant réservés le nom d'une propriété peut ne pas correspondre au nom de l'attribut, il faudra pour l'attribut class utiliser la propriété className par exemple.

var champ = document.getElementById("champ");

champ.setAttribute("class", "champ");
champ.setAttribute("placeholder", "Saisie");
champ.setAttribute("value", "");

// champ
console.log(champ.className);
// Saisie
console.log(champ.placeholder);
// ""
console.log(champ.value);
